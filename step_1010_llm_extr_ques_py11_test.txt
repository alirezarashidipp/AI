import uuid
import ssl
import httpx
import truststore
import openai
import json

from deepeval import assert_test
from deepeval.metrics import GEval
from deepeval.test_case import LLMTestCase
from deepeval.models.base_model import DeepEvalBaseLLM

from token_jwt import get_token
from step_1010_llm_extr_ques_py11_2 import extract_jira_metadata


# ---------------------------------------------------------
# 1. Enterprise OpenAI Client
# ---------------------------------------------------------

BASE_URL = "Link"
USER_ID = "UC1939194"
MODEL = "GPT-4.1"


def get_enterprise_openai_client():

    jwt = get_token()

    ctx = truststore.SSLContext(ssl.PROTOCOL_TLS_CLIENT)

    httpx_client = httpx.Client(
        http2=True,
        verify=ctx
    )

    default_headers = {
        "Content-Type": "application/json",
        "x-correlation-id": str(uuid.uuid4()),
        "x-usersession-id": str(uuid.uuid4())
    }

    client = openai.OpenAI(
        api_key="test",
        http_client=httpx_client,
        base_url=BASE_URL,
        default_headers=default_headers
    )

    return client, jwt


# ---------------------------------------------------------
# 2. DeepEval Judge Model
# ---------------------------------------------------------

class EnterpriseJudgeLLM(DeepEvalBaseLLM):

    def load_model(self):
        return get_enterprise_openai_client()

    def generate(self, prompt: str) -> str:

        client, jwt = self.load_model()

        extra_headers = {
            "X-HSBC-E2E-Trust-Token": jwt
        }

        response = client.chat.completions.create(
            model=MODEL,
            messages=[
                {"role": "system", "content": "You are an evaluation judge."},
                {"role": "user", "content": prompt}
            ],
            user=USER_ID,
            temperature=0,
            max_tokens=500,
            extra_headers=extra_headers
        )

        return response.choices[0].message.content

    async def a_generate(self, prompt: str) -> str:
        return self.generate(prompt)

    def get_model_name(self):
        return "enterprise-gpt-judge"


# ---------------------------------------------------------
# 3. DeepEval Test
# ---------------------------------------------------------

def test_jira_metadata_extraction():

    jira_text = """
    As a Tfx team,
    I want to add a new column in the Project class
    so that we can store and manage additional data related to the project.
    """

    # ---- run your extraction model
    result = extract_jira_metadata(jira_text)

    if result is None:
        raise ValueError("Model returned None")

    result_dict = result.model_dump()
    output_json = json.dumps(result_dict, indent=2)

    print("Model Output:")
    print(output_json)

    # ---- Judge Model
    judge_model = EnterpriseJudgeLLM()

    # ---- Evaluation Metric
    metric = GEval(
        name="Jira Metadata Extraction Quality",
        criteria="""
Evaluate whether the extracted Jira metadata correctly identifies
WHO (actor), WHAT (requested work), WHY (business value),
and Acceptance Criteria if present.
""",
        evaluation_steps=[
            "Check if WHO is correctly identified from the ticket",
            "Check if WHAT describes the requested implementation",
            "Check if WHY explains the business value",
            "Check if acceptance criteria are correctly extracted"
        ],
        model=judge_model
    )

    # ---- Test Case
    test_case = LLMTestCase(
        input=jira_text,
        actual_output=output_json
    )

    # ---- Run DeepEval
    assert_test(test_case, [metric])


# ---------------------------------------------------------
# 4. Run Test
# ---------------------------------------------------------

if __name__ == "__main__":

    test_jira_metadata_extraction()
