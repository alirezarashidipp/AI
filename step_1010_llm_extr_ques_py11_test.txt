import json

from deepeval import assert_test
from deepeval.metrics import GEval
from deepeval.test_case import LLMTestCase

from step_1010_llm_extr_ques_py11_2 import extract_jira_metadata


def test_jira_metadata_extraction():

    jira_text = """
    As a Tfx team,
    I want to add a new column in the Project class
    so that we can store and manage additional data related to the project.
    """

    result = extract_jira_metadata(jira_text)

    # اگر خروجی dict نباشد
    if not isinstance(result, dict):
        result = json.loads(result)

    output_json = json.dumps(result, indent=2)

    metric = GEval(
        name="Jira Agile Metadata Extraction",
        criteria="""
        Evaluate whether the model correctly extracts Agile metadata from the Jira ticket.

        The output must correctly identify:

        WHO:
        - the actor or team performing the action.

        WHAT:
        - the technical implementation action.

        WHY:
        - the business value or purpose.

        CUSTOMER IMPACT:
        - whether user/customer impact is mentioned.

        ACCEPTANCE CRITERIA:
        - whether AC is detected.

        The extracted evidence must correspond to the relevant concept.
        """,
        evaluation_params=["input", "actual_output"]
    )

    test_case = LLMTestCase(
        input=jira_text,
        actual_output=output_json
    )

    assert_test(test_case, [metric])
