import json

from deepeval import assert_test
from deepeval.metrics import GEval
from deepeval.test_case import LLMTestCase

from step_1010_llm_extr_ques_py11_2 import extract_jira_metadata


def test_metadata_extraction():

    jira_text = """
    As a Tfx team,
    I want to add a new column in the Project class
    so that we can store and manage additional data related to the project.
    """

    result = extract_jira_metadata(jira_text)

    # اگر خروجی dict نیست
    if not isinstance(result, dict):
        result = json.loads(result)

    metric = GEval(
        name="Jira Metadata Extraction Quality",
        criteria="""
        Evaluate whether the model correctly extracts structured Agile metadata
        from a Jira ticket description.

        The evaluation should check:

        1. WHO → correct identification of actor/team
        2. WHAT → correct technical implementation intent
        3. WHY → correct business value or motivation
        4. AC presence → whether acceptance criteria are detected
        """,
        evaluation_params=["input", "actual_output"]
    )

    test_case = LLMTestCase(
        input=jira_text,
        actual_output=result
    )

    assert_test(test_case, [metric])
