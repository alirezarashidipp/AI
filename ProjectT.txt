import os
import uuid
import ssl
import httpx
import truststore
import pandas as pd
import openai

from token_jwt import get_token


# =====================================
# 1️⃣ Enterprise LLM Configuration
# =====================================

BASE_URL = "Link"
USER_ID = "UC1939194"
MODEL = "GPT-4.1"


def get_enterprise_client():
    """
    Create enterprise OpenAI client compatible with company gateway
    """
    ctx = truststore.SSLContext(ssl.PROTOCOL_TLS_CLIENT)

    httpx_client = httpx.Client(
        http2=True,
        verify=ctx
    )

    default_headers = {
        "Content-Type": "application/json",
        "x-correlation-id": str(uuid.uuid4()),
        "x-usersession-id": str(uuid.uuid4())
    }

    client = openai.OpenAI(
        api_key="test",
        http_client=httpx_client,
        base_url=BASE_URL,
        default_headers=default_headers
    )

    return client


def call_llm(messages, max_tokens=300):
    """
    Enterprise-safe LLM call
    (creates new JWT each call)
    """

    jwt = get_token()

    client = get_enterprise_client()

    extra_headers = {
        "X-HSBC-E2E-Trust-Token": jwt
    }

    response = client.chat.completions.create(
        model=MODEL,
        messages=messages,
        user=USER_ID,
        extra_headers=extra_headers,
        max_tokens=max_tokens,
        temperature=0
    )

    return response.choices[0].message.content


# =====================================
# 2️⃣ Load Data
# =====================================

df = pd.read_csv("employees.csv")


schema_description = """
Dataset schema:

employee_id: integer, unique identifier
status: string, values = active, inactive, terminated
department: string, department name
current_month: string, format YYYY-MM
"""


# =====================================
# 3️⃣ Generate Pandas Query
# =====================================

def generate_pandas_query(question):

    prompt = f"""
You are a Python data analyst.

A pandas DataFrame called df is available.

Schema:
{schema_description}

User question:
{question}

Write ONLY valid pandas code.
The final result must be stored in a variable called result.

Do not explain anything.
"""

    messages = [
        {"role": "system", "content": "You generate pandas code."},
        {"role": "user", "content": prompt}
    ]

    return call_llm(messages)


# =====================================
# 4️⃣ Execute Query
# =====================================

def execute_query(code):

    local_vars = {"df": df}

    exec(code, {}, local_vars)

    return local_vars.get("result")


# =====================================
# 5️⃣ Generate Human Answer
# =====================================

def generate_human_answer(question, raw_result):

    prompt = f"""
User question:
{question}

Computed result:
{raw_result}

Write a short professional answer in English.
Do not mention pandas or code.
"""

    messages = [
        {"role": "system", "content": "You are a professional data assistant."},
        {"role": "user", "content": prompt}
    ]

    return call_llm(messages)


# =====================================
# 6️⃣ Main Flow
# =====================================

def main():

    print("\n=== Employee Data AI Assistant ===")

    question = input("\nAsk your question: ")

    if not question.strip():
        print("Question cannot be empty")
        return

    try:

        # Step 1: generate pandas code
        pandas_code = generate_pandas_query(question)

        print("\nGenerated pandas code:")
        print(pandas_code)

        # Step 2: execute query
        raw_result = execute_query(pandas_code)

        # Step 3: generate human answer
        final_answer = generate_human_answer(question, raw_result)

        print("\n--- Answer ---")
        print(final_answer)

    except Exception as e:
        print("Error:", str(e))


if __name__ == "__main__":
    main()
